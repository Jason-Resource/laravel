<?php

/**************************************************
	Eloquent 模型

A、
创建模型服务提供者
php artisan make:provider ModelServiceProvider

B、
在 public function register() 方法中注册
$this->app->bind('UserModel', \App\Models\User::class);

C、
$model = app('UserModel');


/**************************************************
	原生
**************************************************/
use DB;

$model = DB::table('permissions');		//permissions是去除了表前缀的表名称；这里返回的对象是Illuminate\Database\Query\Builder，注意：有些方法可能不能使用（比如findOrFail）


/**************************************************
	其他
**************************************************/
use App\Model\Permissions;

$info = Permissions::find(1);

或

$model = app(Permissions::class);
$info = $model->find(1);
